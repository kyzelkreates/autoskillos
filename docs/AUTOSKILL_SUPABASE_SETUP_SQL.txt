2. TABLES

create table if not exists public.autoskill_os_profiles (
  id              uuid primary key default gen_random_uuid(),
  auth_user_id    uuid unique,
  display_name    text not null,
  email           text,
  role            text not null check (role in ('therapist', 'patient')),
  status          text not null default 'active' check (status in ('active', 'inactive', 'archived')),
  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now()
);

create table if not exists public.autoskill_os_patient_links (
  id              uuid primary key default gen_random_uuid(),
  therapist_id    uuid not null references public.autoskill_os_profiles(id) on delete cascade,
  patient_id      uuid not null references public.autoskill_os_profiles(id) on delete cascade,
  status          text not null default 'active' check (status in ('active', 'paused', 'ended')),
  invite_code     text unique,
  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now(),
  unique (therapist_id, patient_id)
);

create table if not exists public.autoskill_os_checkins (
  id                      uuid primary key default gen_random_uuid(),
  patient_id              uuid not null references public.autoskill_os_profiles(id) on delete cascade,
  mood_score              integer check (mood_score between 1 and 10),
  anxiety_score           integer check (anxiety_score between 1 and 10),
  sleep_quality           text check (sleep_quality in ('very_poor', 'poor', 'fair', 'good', 'very_good')),
  energy_score            integer check (energy_score between 1 and 10),
  used_coping_strategy    boolean default false,
  felt_overwhelmed        boolean default false,
  concentration_difficulty boolean default false,
  connected_with_support  boolean default false,
  therapist_note          text,
  extra_support_requested boolean default false,
  support_flag            text not null default 'none' check (support_flag in ('none', 'low', 'medium', 'high', 'needs_review')),
  is_demo                 boolean not null default false,
  submitted_at            timestamptz not null default now(),
  created_at              timestamptz not null default now(),
  updated_at              timestamptz not null default now()
);

create table if not exists public.autoskill_os_lessons (
  id              text primary key,
  module_id       text not null,
  module_title    text not null,
  lesson_number   integer not null,
  title           text not null,
  estimated_time  text,
  summary         text,
  content         jsonb not null default '{}'::jsonb,
  is_active       boolean not null default true,
  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now()
);

create table if not exists public.autoskill_os_lesson_progress (
  id              uuid primary key default gen_random_uuid(),
  patient_id      uuid not null references public.autoskill_os_profiles(id) on delete cascade,
  lesson_id       text not null references public.autoskill_os_lessons(id) on delete cascade,
  status          text not null default 'not_started' check (status in ('not_started', 'in_progress', 'completed')),
  reflection_note text,
  completed_at    timestamptz,
  is_demo         boolean not null default false,
  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now(),
  unique (patient_id, lesson_id)
);

create table if not exists public.autoskill_os_support_flags (
  id              uuid primary key default gen_random_uuid(),
  patient_id      uuid not null references public.autoskill_os_profiles(id) on delete cascade,
  checkin_id      uuid references public.autoskill_os_checkins(id) on delete set null,
  flag_level      text not null check (flag_level in ('low', 'medium', 'high', 'needs_review')),
  title           text not null,
  description     text,
  reviewed_by     uuid references public.autoskill_os_profiles(id) on delete set null,
  reviewed_at     timestamptz,
  status          text not null default 'open' check (status in ('open', 'reviewed', 'closed')),
  is_demo         boolean not null default false,
  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now()
);

create table if not exists public.autoskill_os_patient_notes (
  id              uuid primary key default gen_random_uuid(),
  patient_id      uuid not null references public.autoskill_os_profiles(id) on delete cascade,
  therapist_id    uuid references public.autoskill_os_profiles(id) on delete set null,
  note_type       text not null default 'general' check (note_type in ('general', 'session', 'progress', 'support', 'lesson_reflection')),
  note            text not null,
  visibility      text not null default 'therapist_only' check (visibility in ('therapist_only', 'shared_with_patient', 'patient_private')),
  is_demo         boolean not null default false,
  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now()
);

create table if not exists public.autoskill_os_sync_events (
  id              uuid primary key default gen_random_uuid(),
  source          text not null check (source in ('dashboard', 'patient_pwa')),
  event_type      text not null,
  patient_id      uuid references public.autoskill_os_profiles(id) on delete cascade,
  payload         jsonb not null default '{}'::jsonb,
  sync_status     text not null default 'pending' check (sync_status in ('pending', 'synced', 'failed')),
  error_message   text,
  is_demo         boolean not null default false,
  created_at      timestamptz not null default now(),
  synced_at       timestamptz
);

4. UPDATED_AT TRIGGER FUNCTION
====================================================

create or replace function public.autoskill_os_set_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

drop policy if exists "Profiles can read own profile" on public.autoskill_os_profiles;
create policy "Profiles can read own profile"
  on public.autoskill_os_profiles for select to authenticated
  using (auth_user_id = auth.uid());

drop policy if exists "Profiles can update own profile" on public.autoskill_os_profiles;
create policy "Profiles can update own profile"
  on public.autoskill_os_profiles for update to authenticated
  using (auth_user_id = auth.uid())
  with check (auth_user_id = auth.uid());

drop policy if exists "Trainers/Supervisors can read linked employee profiles" on public.autoskill_os_profiles;
create policy "Trainers/Supervisors can read linked employee profiles"
  on public.autoskill_os_profiles for select to authenticated
  using (
    exists (
      select 1 from public.autoskill_os_patient_links link
      join public.autoskill_os_profiles therapist on therapist.id = link.therapist_id
      where link.patient_id = autoskill_os_profiles.id
        and therapist.auth_user_id = auth.uid()
        and link.status = 'active'
    )
  );
